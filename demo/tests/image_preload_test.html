✅ 推荐方案：图片预加载 + 对象池 
// 1. 页面加载时预加载所有图片
const imageCache = {};
function preloadImages() {
    const symbols = ['WW', 'C1', 'M1', 'M2', 'M3', 'M4', 'M5', 'M6', 'M7'];
    symbols.forEach(symbol => {
        const img = new Image();
        img.src = \`img/\${symbol}.png\`;
        imageCache[symbol] = img;
    });
}

// 2. 更新格子时复用预加载的图片
function updateCellDisplay(cell) {
    const symbolName = getSymbolName(cell.symbol);
    const cachedImg = imageCache[symbolName];
    
    // 复用已加载的图片（克隆或直接使用）
    const img = cachedImg.cloneNode(false);
    cell.element.innerHTML = '';
    cell.element.appendChild(img);
}
📊 预期提升：
  • 首次渲染：慢（需要加载图片）
  • 后续渲染：快 10-50 倍（从缓存读取）
  • 内存占用：增加少量（预加载 9 张图片）
  • 用户体验：显著提升 ⚡
